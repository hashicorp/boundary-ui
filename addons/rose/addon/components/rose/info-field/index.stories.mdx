# Info Field

A information component containing a title, subtitle, description.
Subtitle can be configured with optional icon.

export const Template = (args) => ({
  template: hbs`
      <Rose::InfoField
        @title={{title}}
        @subtitle={{subtitle}}
        @icon={{icon}}
        @description={{description}}
      />
    `,
  context: args,
});
