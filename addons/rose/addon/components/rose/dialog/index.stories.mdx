# Dialog

A dialog component with optional modal capabilities.

export const Template = (args) => ({
  template: hbs`
      <Rose::Dialog
        @modal={{modal}}
        @dismiss={{dismiss}}
        @icon={{icon}}
        @heading={{heading}}
        @dismissButtonText={{dismissButtonText}}
        @style={{style}}
      as |dialog|>
        <dialog.body>
          Dialog Body
        </dialog.body>
        <dialog.footer>
          Footer
        </dialog.footer>
      </Rose::Dialog>
    `,
  context: args,
});
