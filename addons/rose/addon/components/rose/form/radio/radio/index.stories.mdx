# Form Radiofield

Renders a form radiofield.

```handlebars
<Rose::Form::Radio::Radio
  @label='Yes'
  @error={{false}}
  @value='yes-agree'
  @selectedValue='yes-agree'
  @disabled={{false}}
  @changed={{fn}}
/>
```

export const Template = (args) => ({
  template: hbs`
      <Rose::Form::Radio::Radio
        @label={{label}}
        @error={{error}}
        @value={{value}}
        @selectedValue={{selectedValue}}
        @disabled={{disabled}}
        @icon={{icon}}
        @helperText={{helperText}}
      />`,
  context: args,
});
