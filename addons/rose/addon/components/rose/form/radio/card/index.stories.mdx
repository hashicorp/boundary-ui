# Form Radiofield in a wide format

Renders a form radiofield in a Wide format.

```htmlbars
<Rose::Form::Radio::Card
  @label="Yes"
  @error={{false}}
  @value="yes-agree"
  @selectedValue="yes-agree"
  @disabled={{false}}
  @changed={{fn}}
  @layout='wide'
  />
```

<Preview>
  <Story name="Basic Wide" height="10rem">{{
    template: hbs`
      <Rose::Form::Radio::Card
        @label={{label}}
        @error={{error}}
        @value={{value}}
        @selectedValue={{selectedValue}}
        @disabled={{disabled}}
        @icon={{icon}}
        @helperText={{helperText}}
        @layout='wide'
        />
    `,
    context: {
      label: text('text', 'AWS'),
      error: boolean('error', false),

        'value and selectedValue must match to select radiofield'
      ),
      helperText: text('helperText', 'Record financial transactions using'),
      disabled: boolean('disabled', false),
      icon: select(
        'icon',
        {
          'none': null,
          'flight-icons/svg/azure-color-16': 'flight-icons/svg/azure-color-16',

        null
      ),
    }
  }}</Story>
</Preview>
