# Form radiofield group

Renders a form radiofield group.

```handlebars
<Rose::Form::Radio::Group
  @disabled={{false}}
  @name={{name}}
  @selectedValue='disagree'
  @changed={{fn}}
  @inline={{false}}
  as |radioGroup|
>
  <radioGroup.radio @label='Agree' @value='agree' />
  <radioGroup.radio @label='disagree' @value='disagree' />
  <radioGroup.radio @label='maybe' @value='maybe' />
</Rose::Form::Radio::Group>
```

export const Template = (args) => ({
  template: hbs`
      <Rose::Form::Radio::Group
        @disabled={{disabled}}
        @name={{name}}
        @selectedValue={{selectedValue}}
        @changed={{changed}}
        @inline={{inline}}
        as |radioGroup|
      >
        <radioGroup.radio
          @label={{firstFieldLabel}}
          @value={{firstFieldValue}}
          @icon={{firstFieldIcon}}
        />
        <radioGroup.radio @label={{secondFieldLabel}} @value={{secondFieldValue}} />
        <radioGroup.radio @label={{thirdFieldLabel}} @value={{thirdFieldValue}} />
      </Rose::Form::Radio::Group>`,
  context: args,
});
