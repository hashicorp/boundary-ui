# Select (removed)

Rose/Form/Select has been removed in favor of
[HDS::Form::Select](https://design-system-components-hashicorp.vercel.app/components/form/select).

```handlebars
<Hds::Form::Select::Field
  @value={{this.selectedValue}}
  @isInvalid={{false}}
  disabled={{false}}
  name='select_name'
  {{on 'change' (action (mut this.selectedValue) value='target.value')}}
  as |F|
>
  <F.Label>Color</F.Label>
  <F.HelperText>Set your desired color</F.Label>
  <F.Options>
    <option>Choose a color</option>
    <option value='red' selected={{eq this.selectedValue 'red'}}>Red</option>
    <option value='green' selected={{eq this.selectedValue 'green'}}>Green</option>
    <option value='blue' selected={{eq this.selectedValue 'blue'}}>Blue</option>
  </F.Options>
</Hds::Form::Select::Field>
```

export const Template = (args) => ({
  template: hbs`
      <Hds::Form::Select::Field
        @value={{value}}
        @isInvalid={{error}}
        disabled={{disabled}}
        name={{name}}
        {{on 'change' (action (mut value) value='target.value')}}
        as |F|
      >
        <F.Label>{{label}}</F.Label>
        <F.HelperText>{{helperText}}</F.HelperText>
        <F.Options>
          <option>Choose an option</option>
          <option value='value-1' selected={{eq value 'value-1'}}>Value 1</option>
          <option value='value-2' selected={{eq value 'value-2'}}>Value 2</option>
          <option value='value-3' selected={{eq value 'value-3'}}>Value 3</option>
        </F.Options>
      </Hds::Form::Select::Field>`,
  context: args,
});
