# Form Textarea

Renders a simple form textarea.

```handlebars
<Rose::Form::Textarea
  @name='field'
  @value='Descriptive content'
  @label='Description'
  @helperText='Enter a description'
  @error={{false}}
  @disabled={{false}}
  readonly={{false}}
  @link='www.example.net'
  @linkText='link text goes here'
/>
```

export const Template = (args) => ({
  template: hbs`
      <Rose::Form::Textarea
        @name='field'
        @value={{value}}
        @label={{label}}
        @helperText={{helperText}}
        @error={{false}}
        @disabled={{disabled}}
        readonly={{read-only}}
        @link='www.example.net'
        @linkText='link text goes here'
      />`,
  context: args,
});

export const ErrorTemplate = (args) => ({
  template: hbs`
      <Rose::Form::Textarea
        @name='field'
        @value={{value}}
        @label={{label}}
        @helperText={{helperText}}
        @error={{true}}
        @disabled={{disabled}}
        readonly={{read-only}}
        as |field|
      >
        <field.errors as |errors|>
          <errors.message>An error occurred.</errors.message>
        </field.errors>
      </Rose::Form::Textarea>`,
  context: args,
});

export const ContextualTemplate = (args) => ({
  template: hbs`
      <Rose::Form::Textarea
        @value='Text'
        @error={{true}}
        @contextual={{true}}
        as |field|
      >
        <hr />
        <field.label>Label</field.label>
        <hr />
        <field.helperText>Help</field.helperText>
        <hr />
        <field.field />
        <hr />
        <field.errors as |errors|>
          <errors.message>An error occurred.</errors.message>
        </field.errors>
        <hr />
      </Rose::Form::Textarea>`,
  context: args,
});
