# Form Checkbox

Renders a form input checkbox.

```handlebars
<Rose::Form::Checkbox
  @name='boolean-value'
  @label='Yes?'
  @checked={{false}}
  @error={{false}}
  @disabled={{false}}
/>
```

export const Template = (args) => ({
  template: hbs`
      <Rose::Form::Checkbox
        @name={{name}}
        @label={{label}}
        @description={{description}}
        @helperText={{helperText}}
        @checked={{checked}}
        @error={{error}}
        @disabled={{disabled}}
        />`,
  context: args,
});

export const ErrorTemplate = (args) => ({
  template: hbs`
      <Rose::Form::Checkbox
        @name={{name}}
        @label={{label}}
        @checked={{checked}}
        @error={{true}}
        @disabled={{disabled}}
        as |field|
      >
        <field.errors as |errors|>
          <errors.message>An error occurred.</errors.message>
        </field.errors>
      </Rose::Form::Checkbox>`,
  context: args,
});
